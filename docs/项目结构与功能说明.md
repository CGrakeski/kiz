# 项目结构与功能说明
```
- deps
    | - bigint.hpp # 无限精度整数BigInt定义与实现
    | - hashmap.hpp # 键为std::string的高性能HashMap
    | - dict.hpp # 布局为{HashedValue: {KT, VT}}的高性能HashMap(HashedValue类型为BigInt)
    | - decimal.hpp # 无限精度小数Decimal定义与实现
    | - u8str.hpp # Utf8Char, Utf8Str定义与实现

- libs
    | - builtins
         | - include
             | - builtin_lib.hpp # builtins库的初始化函数定义
             | - builtin_methods.hpp # 内置对象的方法定义
             | - builtin_functions.hpp # 内置函数定义
         | - int_methods.cpp # Int类型的方法集
         | - decimal_methods.cpp # Decimal类型的方法集
         | - str_methods.cpp # Str类型的方法集
         | - dict_methods.cpp # Dict类型的方法集
         | - list_methods.cpp # List类型的方法集
         | - bool_methods.cpp # Bool类型的方法集
         | - nil_methods.cpp # Nil类型的方法集
         | - object_methods.cpp # 其他类型的方法集
         | - file_handle_methods.cpp # FileHandle类型的方法集
         | - builtin_functions.cpp # 内置函数
         | - builtin_lib.cpp # 注册builtins库的函数
    | - os
         | - include
             | - os_lib.hpp # os库核心定义
         | - os_lib.cpp # os库核心实现

- src
    | - kiz.hpp # 定义KizStopRunningSingal和NativeFuncError两个错误类和DEBIG_OUTPUT宏
    | - main.cpp # 编译到native入口, 包含控制台参数处理
    | - version.hpp.in # 版本好
    | - wasm_interface.cpp # 编译到wasm编译入口

    | - lexer
         | - lexer.hpp # Lexer核心定义与TokenType定义
         | - lexer.cpp # Lexer核心按状态跳转
         | - read_string.cpp # 解析字符串
         | - read_num.cpp # 解析数字

    | - parser
         | - ast.hpp # ast核心定义与实现
         | - parser.hpp # Parser核心定义
         | - parser.cpp # Parser核心函数
         | - parse_stmt.cpp # 解析语句
         | - parae_expr.cpp # 解析表达式

    | - ir_gen
         | - ir_gen.hpp # IRGenerator核心定义
         | - ir_gen.cpp # IRGenerator核心函数
         | - gen_stmt.cpp # 生产语句的IR
         | - gen_expr.cpp # 生产表达式的IR

    | - opcode
         | - opcode.hpp # 指令操作码定义

    | - models
         | - models.hpp # 运行时堆对象系统的定义与实现

    | - vm
         | - vm.hpp # Instruction/CallFrame/Vm核心定义
         | - vm.cpp # Vm核心函数实现
         | - execuate_unit.cpp # Vm核心执行单元
         | - handle_error.cpp # 处理throw/ensure
         | - handle_import.cpp # 处理import
         | - handle_call.cpp # 处理Bool/call/getattr
         | - entry_builtins.cpp # 注册内置对象
         | - entry_std_modules.cpp # 注册C++实现的标准库

    | - error_reporter
         | - error_reporter.hpp # 错误处理器核心定义
         | - error_reporter.cpp # 错误处理器与TraceBack实现

    | - src_manager
         | - src_manager.hpp # 代码源文件管理器(用于报错时获取文件错误代码处切片)核心定义
         | - src_manager.cpp # 代码源文件管理器核心实现

    | - repl
         | - color.hpp # ASCII控制台颜色控制字符集
         | - repl.hpp # Repl核心定义
         | - repl.cpp # Repl核心实现
         | - repl_readline.cpp # Repl获取用户输入核心

```
